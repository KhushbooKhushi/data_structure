/* 

max_element(v.begin(), v.end())

Returns an iterator pointing to the largest element.
Dereference it (*) to get the value.

*/


class Solution{
public:
	void vector<bool>kidsWithCandies(vector<int>& candies, int extraCandies){
		int max = *max_element(candies.begin(), candies.end());

		vector<bool> ans;
		for(int i=0;i<candies.size();i++)
		{
			if((extraCandies+candies[i]) >= max)
				ans.push_back(true);
			else
				ans.push_back(false);
		}
	
		return ans;
	}
};
