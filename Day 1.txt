#include <iostream>
using namespace std;

int add(int a, int b){
    //return a+b;
    
    if(b>=0){
        while(--b){
            a++;
        }
    }
    else{
        while(b++){
            a--;
        }
    }
    return a;
}

     int result = add(5,6);
     cout<<result;

    return 0;
}
