/* 1281. Subtract the Product and Sum of Digits of an Integer */

class Solution {
public:
    int subtractProductAndSum(int n) {
        int sum=0, mul=1;
        while(n>0){
            sum += n%10;
            mul *= n%10;
            n = n/10;
        }
        return mul - sum;
    }
};


/* using string conversion */

#include <iostream>
using namespace std;

int main() {
    int num = 453;
   string s = to_string(num);
   int sum=0,mul=1;
   for(char c : s){
       int d = c - '0';
       sum+=d;
       mul*=d;
   }
    cout<<sum<<endl;
    cout<<mul<<endl;
    return 0;
}
