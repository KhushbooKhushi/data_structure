/* 1523 Count Odd Numbers in an Interval Range  */

//Number of odd numbers from 1 to n  = (n+1)/2;


class Solution {
public:
    int countOdds(int low, int high) {
        return (high+1)/2 - (low/2);
    }
};




class Solution {
public:
    int countOdds(int low, int high) {
         int count=0;

         for(int i=low;i<=high;i++){
            if(i & 1)
            count++;
         }
         return count;
    }
};
