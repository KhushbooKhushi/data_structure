/* 344. Reverse String */ 


class Solution {
public:
    void reverseString(vector<char>& s) {
        int first=0,last=s.size()-1;
        while(first<=last){
            
            char c = s[first];
            s[first] = s[last];
            s[last] = c;
            
            first++;
            last--;
        }
    }
};
